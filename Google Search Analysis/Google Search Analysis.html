# Finding out how the total searches for the phrase we
# are looking into has changed over time

# Connecting to Google and building the payload
data = TrendReq(hl='en-US', tz=360)
data.build_payload(kw_list=['Cyber Security'])
data = data.interest_over_time()

figure, axes = plt.subplots(figsize=(10, 10))
data['Cyber Security'].plot()
plt.style.use('seaborn-dark-palette')
plt.title('Total Google Searches for Cyber Security', fontweight='bold')
plt.xlabel('Year')
plt.ylabel('Total Count')
plt.show()
